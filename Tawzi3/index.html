            // عرض السنة في التذييل

            document.getElementById('year').textContent = new Date().getFullYear();

            

            // تحميل البيانات عند فتح الصفحة

            loadStudents();

            

            // أحداث الأزرار

            searchBtn.addEventListener('click', filterStudents);

            autoAssignBtn.addEventListener('click', autoAssignStudents);

            clearAssignmentsBtn.addEventListener('click', clearAssignments);

            saveBtn.addEventListener('click', saveAssignments);

            

            // دالة لتحميل الطلبة من قاعدة البيانات

            function loadStudents() {

                loadingSpinner.style.display = 'flex';

                

                database.ref('students').on('value', (snapshot) => {

                    allStudents = [];

                    snapshot.forEach(childSnapshot => {

                        const student = childSnapshot.val();

                        student.id = childSnapshot.key;

                        allStudents.push(student);

                    });

                    

                    updateStatistics();

                    renderGroups();

                    loadingSpinner.style.display = 'none';

                }, (error) => {

                    console.error('Error loading students:', error);

                    loadingSpinner.style.display = 'none';

                    alert('حدث خطأ أثناء تحميل بيانات الطلبة');

                });

            }

            

            // دالة لتحديث الإحصائيات

            function updateStatistics() {

                const total = allStudents.length;

                const assigned = allStudents.filter(s => s.group).length;

                const unassigned = total - assigned;

                

                totalStudentsEl.textContent = total;

                assignedStudentsEl.textContent = assigned;

                unassignedStudentsEl.textContent = unassigned;

            }

            

            // دالة لعرض الأفواج والطلبة

            function renderGroups() {

                groupsContainer.innerHTML = '';

                

                // عرض الأفواج الستة

                groups.forEach(group => {

                    const groupStudents = allStudents.filter(s => s.group === group.id);

                    createGroupCard(group, groupStudents);

                });

                

                // عرض الطلبة غير الموزعين

                const unassignedStudents = allStudents.filter(s => !s.group);

                createGroupCard(

                    { id: 'unassigned', name: 'طلبة غير موزعين', time: '' },

                    unassignedStudents

                );

            }

            

            // دالة لإنشاء بطاقة الفوج

            function createGroupCard(group, students) {

                const groupCard = document.createElement('div');

                groupCard.className = 'group-card';

                groupCard.dataset.groupId = group.id;

                

                const groupHeader = document.createElement('div');

                groupHeader.className = 'group-header';

                

                const groupTitle = document.createElement('div');

                groupTitle.className = 'group-title';

                groupTitle.textContent = group.name;

                

                const groupInfo = document.createElement('div');

                groupInfo.className = 'group-info';

                

                if (group.time) {

                    const groupTime = document.createElement('div');

                    groupTime.className = 'group-time';

                    groupTime.textContent = group.time;

                    groupInfo.appendChild(groupTime);

                }

                

                const groupCount = document.createElement('div');

                groupCount.className = 'group-count';

                groupCount.textContent = students.length;

                groupInfo.appendChild(groupCount);

                

                groupHeader.appendChild(groupTitle);

                groupHeader.appendChild(groupInfo);

                

                const studentList = document.createElement('ul');

                studentList.className = 'student-list';

                

                students.forEach(student => {

                    const studentItem = document.createElement('li');

                    studentItem.className = 'student-item';

                    studentItem.dataset.studentId = student.id;

                    

                    const studentName = document.createElement('div');

                    studentName.className = 'student-name';

                    studentName.textContent = `${student.firstName} ${student.lastName}`;

                    

                    const studentId = document.createElement('div');

                    studentId.className = 'student-id';

                    studentId.textContent = student.registrationNumber;

                    

                    const studentActions = document.createElement('div');

                    studentActions.className = 'student-actions';

                    

                    // إضافة أزرار التحريك بين الأفواج

                    if (group.id !== 'unassigned') {

                        const moveBtn = document.createElement('button');

                        moveBtn.className = 'btn-move';

                        moveBtn.innerHTML = '<i class="fas fa-exchange-alt"></i>';

                        moveBtn.title = 'نقل إلى فوج آخر';

                        moveBtn.addEventListener('click', (e) => {

                            e.stopPropagation();

                            showMoveOptions(student.id);

                        });

                        studentActions.appendChild(moveBtn);

                    }

                    

                    studentItem.appendChild(studentName);

                    studentItem.appendChild(studentId);

                    studentItem.appendChild(studentActions);

                    studentList.appendChild(studentItem);

                });

                

                groupCard.appendChild(groupHeader);

                groupCard.appendChild(studentList);

                groupsContainer.appendChild(groupCard);

            }

            

            // دالة لعرض خيارات نقل الطالب

            function showMoveOptions(studentId) {

                currentStudentToMove = studentId;

                const student = allStudents.find(s => s.id === studentId);

                if (!student) return;

                

                const currentGroupId = student.group || 'unassigned';

                moveOptionsList.innerHTML = '';

                

                // إضافة خيار "غير موزع"

                if (currentGroupId !== 'unassigned') {

                    const unassignedOption = document.createElement('div');

                    unassignedOption.className = 'move-option';

                    unassignedOption.textContent = 'غير موزع';

                    unassignedOption.addEventListener('click', () => {

                        moveStudent(studentId, 'unassigned');

                        moveOptions.style.display = 'none';

                    });

                    moveOptionsList.appendChild(unassignedOption);

                }

                

                // إضافة خيارات الأفواج

                groups.forEach(group => {

                    if (group.id !== currentGroupId) {

                        const groupOption = document.createElement('div');

                        groupOption.className = 'move-option';

                        groupOption.textContent = `${group.name} (${group.time})`;

                        groupOption.addEventListener('click', () => {

                            moveStudent(studentId, group.id);

                            moveOptions.style.display = 'none';

                        });

                        moveOptionsList.appendChild(groupOption);

                    }

                });

                

                moveOptions.style.display = 'flex';

            }

            

            // إغلاق نافذة الخيارات عند النقر خارجها

            moveOptions.addEventListener('click', (e) => {

                if (e.target === moveOptions) {

                    moveOptions.style.display = 'none';

                }

            });

            

            // دالة لنقل الطالب بين الأفواج

            function moveStudent(studentId, targetGroupId) {

                const studentIndex = allStudents.findIndex(s => s.id === studentId);

                if (studentIndex === -1) return;

                

                allStudents[studentIndex].group = targetGroupId === 'unassigned' ? null : targetGroupId;

                

                updateStatistics();

                renderGroups();

            }

            

            // دالة للبحث والتصفية

            function filterStudents() {

                const searchTerm = searchInput.value.toLowerCase();

                const selectedGroup = groupFilter.value;

                

                filteredStudents = allStudents.filter(student => {

                    const matchesSearch = searchTerm === '' || 

                        student.firstName.toLowerCase().includes(searchTerm) || 

                        student.lastName.toLowerCase().includes(searchTerm) ||

                        student.registrationNumber.toLowerCase().includes(searchTerm);

                    

                    const matchesGroup = selectedGroup === '' || 

                        (selectedGroup === 'unassigned' && !student.group) || 

                        student.group === selectedGroup;

                    

                    return matchesSearch && matchesGroup;

                });

                

                // عرض نتائج البحث

                renderFilteredResults();

            }

            

            // دالة لعرض نتائج البحث

            function renderFilteredResults() {

                groupsContainer.innerHTML = '';

                

                if (filteredStudents.length === 0) {

                    groupsContainer.innerHTML = '<p style="grid-column: 1/-1; text-align: center;">لا توجد نتائج مطابقة للبحث</p>';

                    return;

                }

                

                const searchResultsGroup = {

                    id: 'searchResults',

                    name: 'نتائج البحث',

                    time: ''

                };

                

                createGroupCard(searchResultsGroup, filteredStudents);

            }

            

            // دالة للتوزيع التلقائي

            function autoAssignStudents() {

                if (!confirm('هل أنت متأكد من أنك تريد توزيع الطلبة تلقائياً؟ سيتم إعادة تعيين التوزيع الحالي.')) {

                    return;

                }

                

                // تصفية الطلبة غير الموزعين

                const unassignedStudents = allStudents.filter(s => !s.group);

                

                if (unassignedStudents.length === 0) {

                    alert('لا يوجد طلبة غير موزعين');

                    return;

                }

                

                // توزيع الطلبة بالتساوي على الأفواج

                const studentsPerGroup = Math.ceil(unassignedStudents.length / groups.length);

                

                groups.forEach((group, index) => {

                    const start = index * studentsPerGroup;

                    const end = start + studentsPerGroup;

                    const groupStudents = unassignedStudents.slice(start, end);

                    

                    groupStudents.forEach(student => {

                        const studentIndex = allStudents.findIndex(s => s.id === student.id);

                        if (studentIndex !== -1) {

                            allStudents[studentIndex].group = group.id;

                        }

                    });

                });

                

                updateStatistics();

                renderGroups();

                alert('تم توزيع الطلبة تلقائياً على الأفواج');

            }

            

            // دالة لإعادة تعيين التوزيع

            function clearAssignments() {

                if (!confirm('هل أنت متأكد من أنك تريد إزالة توزيع جميع الطلبة؟')) {

                    return;

                }

                

                allStudents.forEach(student => {

                    student.group = null;

                });

                

                updateStatistics();

                renderGroups();

                alert('تم إعادة تعيين توزيع الطلبة');

            }

            

            // دالة لحفظ التوزيع في قاعدة البيانات

            function saveAssignments() {

                loadingSpinner.style.display = 'flex';

                

                const updates = {};

                allStudents.forEach(student => {

                    updates[`students/${student.id}/group`] = student.group || null;

                });

                

                database.ref().update(updates)

                    .then(() => {

                        loadingSpinner.style.display = 'none';

                        alert('تم حفظ توزيع الطلبة بنجاح');

                    })

                    .catch(error => {

                        loadingSpinner.style.display = 'none';

                        console.error('Error saving assignments:', error);

                        alert('حدث خطأ أثناء حفظ التوزيع');

                    });

            }

        });
